/* eslint-disable @typescript-eslint/no-explicit-any */
"use client";

"use client";

import { useState, useEffect } from "react";
import DashboardLayout from "@/components/DashboardLayout";
import ProtectedRoute from "@/components/auth/ProtectedRoute";
import { useAuth } from "@/contexts/AuthContext";
import { useRouter } from "next/navigation";
import ProfileAvatar from "@/components/ProfileAvatar";
import { useHealthStatus } from "@/hooks/useHealthStatus";
import { useRecentActivities } from "@/hooks/useRecentActivities";
import { useTranslations } from "next-intl";
import {
    WrenchScrewdriverIcon,
    ClipboardDocumentListIcon,
    ClockIcon,
    CheckCircleIcon,
    ExclamationTriangleIcon,
    CogIcon,
    CpuChipIcon,
    SignalIcon,
    ChartBarIcon
} from '@heroicons/react/24/outline';
import { apiService } from "@/services/api";

interface OperatorStats {
    activeWorkOrders: number;
    completedToday: number;
    pendingTasks: number;
    machinesAssigned: number;
}

export default function OperatorDashboard() {
    const tOperator = useTranslations("dashboard.operator");
    const tHealth = useTranslations("health");
    const tCommon = useTranslations("common");
    const { user, isLoading: authLoading } = useAuth();
    const router = useRouter();
    const { health } = useHealthStatus();
    const { activities, loading: activitiesLoading } = useRecentActivities(4);
    const [stats, setStats] = useState<OperatorStats>({
        activeWorkOrders: 0,
        completedToday: 0,
        pendingTasks: 0,
        machinesAssigned: 0
    });
    const [loading, setLoading] = useState(true);

    // Helper function to get status styling
    const getStatusStyle = (status: string) => {
        switch (status) {
            case 'ONLINE':
                return {
                    dotClass: 'online',
                    badgeClass: 'online'
                };
            case 'WARNING':
                return {
                    dotClass: 'warning',
                    badgeClass: 'warning'
                };
            case 'OFFLINE':
            case 'ERROR':
                return {
                    dotClass: 'offline',
                    badgeClass: 'offline'
                };
            default:
                return {
                    dotClass: '',
                    badgeClass: ''
                };
        }
    };

    // Helper function to get activity styling
    const getActivityStyle = (activity: any) => {
        switch (activity.icon) {
            case 'check':
                return {
                    icon: CheckCircleIcon,
                    bgColor: 'bg-green-50',
                    iconColor: 'text-green-500'
                };
            case 'wrench':
                return {
                    icon: WrenchScrewdriverIcon,
                    bgColor: 'bg-blue-50',
                    iconColor: 'text-blue-500'
                };
            case 'clock':
                return {
                    icon: ClockIcon,
                    bgColor: 'bg-blue-50',
                    iconColor: 'text-blue-500'
                };
            case 'alert':
                return {
                    icon: ExclamationTriangleIcon,
                    bgColor: 'bg-yellow-50',
                    iconColor: 'text-yellow-500'
                };
            default:
                return {
                    icon: ClockIcon,
                    bgColor: 'bg-gray-50',
                    iconColor: 'text-gray-500'
                };
        }
    };

    // Helper function to format relative time
    const getRelativeTime = (date: Date) => {
        const now = new Date();
        const diffInMs = now.getTime() - date.getTime();
        const diffInHours = Math.floor(diffInMs / (1000 * 60 * 60));
        const diffInDays = Math.floor(diffInHours / 24);

        if (diffInHours < 1) {
            return tCommon("time.justNow");
        } else if (diffInHours < 24) {
            return tCommon("time.hoursAgo", { count: diffInHours });
        } else {
            return tCommon("time.daysAgo", { count: diffInDays });
        }
    };

    async function loadOperatorStats() {
        try {
            // Get work orders assigned to this operator
            const workOrdersResponse = await apiService.getWorkOrders();
            const workOrders = workOrdersResponse.data;
            const userWorkOrders = workOrders.filter((order: any) =>
                order.assigned_to === user?._id || order.assigned_to === user?.nom_complet
            );

            const activeOrders = userWorkOrders.filter((order: any) =>
                order.status === 'in_progress' || order.status === 'pending'
            );

            const completedToday = userWorkOrders.filter((order: any) => {
                if (order.status !== 'completed') return false;
                const completedDate = new Date(order.updated_at || order.created_at);
                const today = new Date();
                return completedDate.toDateString() === today.toDateString();
            });

            // Get machines this operator is assigned to (simplified - in real app would have assignment logic)
            const machinesResponse = await apiService.getMachines();
            const machines = machinesResponse.data;
            const assignedMachines = machines.filter((machine: any) =>
                machine.status === 'active' // Simplified logic
            );

            setStats({
                activeWorkOrders: activeOrders.length,
                completedToday: completedToday.length,
                pendingTasks: activeOrders.length, // Simplified
                machinesAssigned: assignedMachines.length
            });
        } catch (error) {
            console.error('Error loading operator stats:', error);
        } finally {
            setLoading(false);
        }
    }

    useEffect(() => {
        if (authLoading) {
            return;
        }

        if (!user) {
            router.replace('/auth/login');
            return;
        }

        // Redirect if not operator
        if (user.role !== 'operator') {
            router.replace('/');
            return;
        }

        // eslint-disable-next-line react-hooks/set-state-in-effect
        loadOperatorStats();
    }, [user, authLoading, router]);


    if (loading) {
        return (
            <div className="min-h-screen flex items-center justify-center bg-gray-50">
                <div className="animate-spin rounded-full h-32 w-32 border-b-2 border-blue-600"></div>
            </div>
        );
    }

    return (
        <ProtectedRoute>
            <DashboardLayout title={tOperator("title")}>
                <div className="bento-grid">
                    {/* Welcome Message */}
                    <div className="col-span-full mb-6 bento-item">
                        <div className="panel">
                            <div className="flex items-center gap-3">
                              
                                <div>
                                    <div className="card-title">{tOperator("welcome", { name: user?.nom_complet || "" })}</div>
                                </div>
                            </div>
                        </div>
                    </div>

                    {/* Stats Cards Row */}
                    <div className="stats-grid">
                        <div className="featured-card">
                            <div className="card-title">{tOperator("stats.activeWorkOrders")}</div>
                            <div className="flex items-center justify-between">
                                <div>
                                    <div className="text-4xl font-bold mb-2">{stats.activeWorkOrders}</div>
                                    <div className="text-blue-200 text-sm">{tOperator("stats.assignedToYou")}</div>
                                </div>
                                <ClipboardDocumentListIcon className="w-16 h-16 text-blue-200 opacity-80" />
                            </div>
                        </div>
                        <div className="panel group">
                            <div className="card-title">{tOperator("stats.completedToday")}</div>
                            <div className="flex items-center justify-between">
                                <div className="flex items-center">
                                    <div className="text-3xl font-bold text-slate-800">{stats.completedToday}</div>
                                </div>
                                <CheckCircleIcon className="w-12 h-12 text-green-600 card-icon" />
                            </div>
                            <div className="text-slate-500 text-sm mt-1">{tOperator("stats.tasksFinishedToday")}</div>
                        </div>
                        <div className="panel group">
                            <div className="card-title flex items-center">
                                <div className="pulse-dot"></div>
                                {tOperator("stats.pendingTasks")}
                            </div>
                            <div className="flex items-center justify-between">
                                <div>
                                    <div className="text-3xl font-bold text-slate-800 mb-1">{stats.pendingTasks}</div>
                                    <div className="text-amber-600 text-sm">{tOperator("stats.awaitingAction")}</div>
                                </div>
                                <ClockIcon className="w-12 h-12 text-amber-600 card-icon" />
                            </div>
                        </div>
                        <div className="panel group relative">
                            <div className="success-badge">
                                <CogIcon className="w-3 h-3 text-white" />
                            </div>
                            <div className="card-title">{tOperator("stats.machinesAssigned")}</div>
                            <div className="flex items-center justify-between">
                                <div>
                                    <div className="text-3xl font-bold text-slate-800 mb-1">{stats.machinesAssigned}</div>
                                    <div className="text-purple-600 text-sm">{tOperator("stats.underYourCare")}</div>
                                </div>
                                <CogIcon className="w-12 h-12 text-purple-600 card-icon" />
                            </div>

                        </div>


                    </div>


                    {/* Quick Actions */}
                    <div className="bento-item panel" style={{ gridColumn: '1 / 3' }}>
                        <div className="card-title">{tOperator("quickActions.title")}</div>
                        <div className="grid grid-cols-1 gap-4 sm:grid-cols-2">
                            <button
                                onClick={() => router.push('/work-orders')}
                                className="inline-flex items-center px-4 py-3 border border-transparent text-sm font-medium rounded-md text-white bg-blue-600 hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-colors"
                            >
                                <ClipboardDocumentListIcon className="h-5 w-5 mr-2" />
                                {tOperator("quickActions.viewWorkOrders")}
                            </button>

                            <button
                                onClick={() => router.push('/machines')}
                                className="inline-flex items-center px-4 py-3 border border-gray-300 text-sm font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-colors"
                            >
                                <CogIcon className="h-5 w-5 mr-2" />
                                {tOperator("quickActions.checkMachines")}
                            </button>

                            <button
                                onClick={() => router.push('/capteurs')}
                                className="inline-flex items-center px-4 py-3 border border-gray-300 text-sm font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-colors"
                            >
                                <CpuChipIcon className="h-5 w-5 mr-2" />
                                {tOperator("quickActions.monitorSensors")}
                            </button>

                            <button
                                onClick={() => router.push('/catalogues')}
                                className="inline-flex items-center px-4 py-3 border border-gray-300 text-sm font-medium rounded-md text-gray-700 bg-white hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 transition-colors"
                            >
                                <WrenchScrewdriverIcon className="h-5 w-5 mr-2" />
                                {tOperator("quickActions.browseCatalogues")}
                            </button>
                        </div>
                    </div>


                    {/* Recent Activity */}
                    <div className="col-span-full bento-item panel">
                        <div className="card-title">{tOperator("recentActivity.title")}</div>
                        <div className="space-y-4">
                            {activitiesLoading ? (
                                <div className="flex items-center justify-center py-8">
                                    <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-600"></div>
                                </div>
                            ) : activities.length > 0 ? (
                                activities.map((activity) => {
                                    const style = getActivityStyle(activity);
                                    const IconComponent = style.icon;
                                    return (
                                        <div key={activity.id} className={`flex items-center space-x-3 p-3 ${style.bgColor} rounded-lg`}>
                                            <div className="flex-shrink-0">
                                                <IconComponent className={`h-5 w-5 ${style.iconColor}`} />
                                            </div>
                                            <div className="min-w-0 flex-1">
                                                <p className="text-sm text-gray-900">{activity.title}</p>
                                                <p className="text-xs text-gray-500">{activity.description}</p>
                                                <p className="text-xs text-gray-400">{getRelativeTime(activity.timestamp ? new Date(activity.timestamp) : new Date())}</p>
                                            </div>
                                        </div>
                                    );
                                })
                            ) : (
                                <div className="text-center py-8 text-gray-500">
                                    <p className="text-sm">{tOperator("recentActivity.empty")}</p>
                                </div>
                            )}
                        </div>
                    </div>
                </div>
            </DashboardLayout>
        </ProtectedRoute>
    );
}
